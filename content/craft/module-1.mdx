  <CompareGrid left='{"title": "Product Manager", "items": ["Owns the <em>why</em> and <em>what</em>", "Defines problems, not just tasks", "Accountable for outcomes, not outputs", "Works backward from user value", "Comfortable with ambiguity"]}' right='{"title": "Project Manager", "items": ["Owns the <em>how</em> and <em>when</em>", "Coordinates execution across teams", "Accountable for timelines and delivery", "Manages dependencies and risk", "Keeps processes running smoothly"]}' />
